#include <iostream>
#include <cstring>
using namespace std;
const int N = 10010,S = 55,M = 1000010;
int n;
int tr[N * S][26],cnt[N * S],idx;
int q[N * S],ne[N * S];
string str;
void insert (string s) {  //Trie树的基本操作
    int p = 0;
    for (int i = 0;i < s.size ();i++) {
        int x = s[i] - 'a';
        if (!tr[p][x]) tr[p][x] = ++idx;
        p = tr[p][x];
    }
    cnt[p]++;
}
void get_next () {
    int hh = 0,tt = -1;  //由于Trie是一棵树，所以要用BFS
    for (int i = 0;i < 26;i++) {
        if (tr[0][i]) q[++tt] = tr[0][i]; //这里是把原来KMP中的0，1号点加入队列
    }
    while (hh <= tt) {
        int t = q[hh++];
        for (int i = 0;i < 26;i++) {
            int &p = tr[t][i];  
            /*
            if (p) {
                int j = ne[t];
                while (j && !tr[j][i]) j = ne[j];  //一直往前跳，知道某一个字符串的后一个字符是i
                if (tr[j][i]) j = tr[j][i];  //对应KMP的if (s[i] == p[j + 1]) j++;
                ne[p] = j;  //求出next得知
                q[++tt] = p;
            }
            */
            // 优化思路 在没有匹配时 把while循环多次跳 优化为 直接跳到ne指针最终跳到的位置
            // 假定在循环第i层时，前i-1层都求对了
            // 在第i层没找到字母j,那么去第i-1层父结点t的next指针的位置就是它最终应该跳到的位置
            if (!p) p = tr[ne[t]][i];  //直接跳到父节点的next指针的第i个儿子
            else {
                ne[p] = tr[ne[t]][i];  //求出next的值
                q[++tt] = p;
            }
        }
    }
}
int main () {
    int T;
    cin >> T;
    while (T--) {
        memset (tr,0,sizeof (tr));
        memset (cnt,0,sizeof (cnt));
        memset (ne,0,sizeof (ne));
        cin >> n;
        for (int i = 1;i <= n;i++) {
            string x;
            cin >> x;
            insert (x);
        }
        cin >> str;
        get_next ();
        int ans = 0;
        for (int i = 0,j = 0;i < str.size ();i++) {
            int t = str[i] - 'a';
            /*
            while (j && !tr[j][t]) j = ne[j];
            if (tr[j][t]) j = tr[j][t];
            int p = j;
            */
            int p = j = tr[j][t];
            while (p) {
                ans += cnt[p];  //这里要加上所有子序列的串，比如she和he都要算
                cnt[p] = 0;
                p = ne[p];
            }
        }
        cout << ans << endl;
    }
    return 0;
}
