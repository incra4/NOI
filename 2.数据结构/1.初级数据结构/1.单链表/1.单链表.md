# 单链表
## [题目链接](https://www.acwing.com/problem/content/828/)

## 标签
单链表

## 代码
```cpp
#include <iostream>
using namespace std;
const int N = 100010;
int n;
int front = -1,idx = 0;
int ne[N],e[N];
void push_front (int x) {
	 e[idx] = x;
	 ne[idx] = front;
	 front = idx;
	 idx++;
}
void push (int k,int x) {
	e[idx] = x;
	ne[idx] = ne[k];
	ne[k] = idx++;
}
void pop (int k) {
	ne[k] = ne[ne[k]];
}
int main () {
	cin >> n;
	while (n--) {
		char ch;
		int x,y;
		cin >> ch;
		if (ch == 'H') {
			cin >> x;
			push_front (x);
		}
		else if (ch == 'I') {
			cin >> x >> y;
			push (x-1,y);
		}
		else {
			cin >> x;
			if (x == 0) front = ne[front];
			pop (x-1);
		}
	}
	for (int i = front;i != -1;i = ne[i]) cout << e[i] << ' ';
    return 0;
}
```
