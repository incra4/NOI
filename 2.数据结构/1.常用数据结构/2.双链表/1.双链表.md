# 双链表
## [题目链接](https://www.acwing.com/problem/content/829/)

**时间复杂度** $O(1)$

**空间复杂度** $O(n)$

## 代码
```cpp
#include <iostream>
using namespace std;
const int N = 100010;
int n;
int e[N],l[N],r[N],idx;
void init () {
	r[0] = 1;
	l[1] = 0;
	idx = 2; 
}
void push_front (int x) {
	l[idx] = 0;
	r[idx] = r[0];
	r[0] = idx;
	l[r[idx]] = idx;
	e[idx] = x;
	idx++;
}
void push_back (int x) {
	r[idx] = 1;
	l[idx] = l[1];
	r[l[idx]] = idx;
	l[1] = idx;
	e[idx] = x;
	idx++;
}
void push_left (int k,int x) {
	l[idx] = l[k];
	r[idx] = k;
	r[l[idx]] = idx;
	l[k] = idx;
	e[idx] = x;
	idx++;
}
void push_right (int k,int x) {
	r[idx] = r[k];
	l[idx] = k;
	l[r[idx]] = idx;
	r[k] = idx;
	e[idx] = x;
	idx++;
}
void pop (int k) {
	r[l[k]] = r[k];
	l[r[k]] = l[k];
}
int main () {
	init ();
	cin >> n;
	while (n--) {
		int k,x;
		string str;
		cin >> str;
		if(str == "L") {
			cin >> x;
			push_front (x);
		}
		else if(str == "R") {
			cin >> x;
			push_back (x);
		}
		else if(str == "D") {
			cin >> k;
			pop (k+1);
		}
		else if(str == "IL") {
			cin >> k >> x;
			push_left (k+1,x);
		}
		else {
			cin >> k >> x;
			push_right (k+1,x);
		}
	}
	for (int i = r[0];i != 1;i = r[i]) cout << e[i] << " ";
	cout << endl;
	return 0;
}
```
