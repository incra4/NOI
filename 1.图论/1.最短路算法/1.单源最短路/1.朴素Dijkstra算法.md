# 朴素Dijkstra算法
## [**题目链接**](https://www.acwing.com/problem/content/851)

**时间复杂度** $O(n^2)$

**空间复杂度** $O(n)$

## 代码
```cpp
#include <iostream>
#include <cstring>
using namespace std;
const int N = 510,INF = 0x3f3f3f3f;
int n,m;
int g[N][N];
int dist[N];
bool st[N];
int dijkstra () {
    memset (dist,0x3f,sizeof (dist));
    dist[1] = 0;
    for (int i = 1;i <= n;i++) {
        int t = -1;
        for (int j = 1;j <= n;j++) {
            if (!st[j] && (t == -1 || dist[t] > dist[j])) t = j;
        }
        st[t] = true;
        for (int j = 1;j <= n;j++) dist[j] = min (dist[j],dist[t]+g[t][j]);
    }
    return dist[n];
}
int main () {
    memset (g,0x3f,sizeof (g));
    cin >> n >> m;
    while (m--) {
        int a,b,c;
        cin >> a >> b >> c;
        g[a][b] = min (g[a][b],c);
    }
    int ans = dijkstra ();
    if (ans == INF) puts ("-1");
    else cout << ans << endl;
    return 0;
}
```
