# SPFA算法求最短路
## [**题目链接**](https://www.acwing.com/problem/content/853/)

**时间复杂度** $O(km)$， $k$ **最坏情况下为** $n$

**空间复杂度** $O(n)$

## 代码
```cpp
#include <iostream>
#include <cstring>
#include <queue>
#include <algorithm>
using namespace std;
const int N = 100010,M = 100010,INF = 0x3f3f3f3f;
int n,m;
int h[N],e[M],w[M],ne[M],idx;
int dist[N];
bool st[N];
void add (int a,int b,int c) {
    e[idx] = b;
    w[idx] = c;
    ne[idx] = h[a];
    h[a] = idx++;
}
int spfa () {
    memset (dist,0x3f,sizeof (dist));
    dist[1] = 0;
    queue <int> q;
    q.push (1);
    st[1] = false;
    while (q.size ()) {
        int t = q.front ();
        q.pop ();
        st[t] = false;
        for (int i = h[t];~i;i = ne[i]) {
            int j = e[i];
            if (dist[j] > dist[t] + w[i]) {
                dist[j] = dist[t] + w[i];
                if (!st[j]) {
                    q.push (j);
                    st[j] = true;
                }
            }
        }
    }
    return dist[n];
}
int main () {
    memset (h,-1,sizeof (h));
    cin >> n >> m;
    while (m--) {
        int a,b,c;
        cin >> a >> b >> c;
        add (a,b,c);
    }
    int ans = spfa ();
    if (ans > INF / 2) puts ("impossible");
    else cout << ans << endl;
    return 0;
}
```
